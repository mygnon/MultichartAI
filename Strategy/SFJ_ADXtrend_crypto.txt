{===============================================================================
  SFJ_ADXtrend_crypto
  -------------------------------------------------------------------------------
  PowerLanguage (EasyLanguage / MultiCharts64) trading strategy for crypto.
  Long + Short. DMI directional crossover gated by an ADX trend-strength filter,
  with an ATR chandelier trailing-stop exit.

  Distinct archetype from the other Strategy-folder systems:
    - Direction comes from the DMI (DI+ vs DI-) crossover (Welles Wilder
      Directional Movement), NOT from a price-channel/band breakout, a pullback,
      a raw ROC, or a SAR flip.
    - The ADX > ADXThresh filter is the key anti-whipsaw gate: a DI cross is only
      taken when trend STRENGTH is high. Pure DI crosses (like MACD zero-crosses
      and PSAR flips) whipsaw on crypto; requiring strong ADX filters the chop.
    - ATR chandelier trailing exit (the exit that worked across crypto trend
      strategies); flat-only entry (no always-in flipping).
  Sizing: _Crypto1MUSD = Round(1,000,000 / Close, 0) ~ $1M notional per trade.

  Inputs (all optimizable; names match the optimizer's ParamAxis exactly):
    DMILen    - DMI / ADX period
    ADXThresh - minimum ADX required to take a DI crossover (trend-strength gate)
    ATRMult   - trailing-stop distance in ATR(14) multiples

  Logic:
    diPlus=DMIPlus(DMILen); diMinus=DMIMinus(DMILen); adxv=ADX(DMILen)
    Long  entry : flat AND adxv>ADXThresh AND diPlus cross over diMinus  -> Buy
    Short entry : flat AND adxv>ADXThresh AND diMinus cross over diPlus   -> SellShort
    Exit  : ATR chandelier trail (ATRMult * ATR(14)) under/over running extreme

  Defaults: DMILen=14, ADXThresh=25, ATRMult=3.0
  Author: generated for Tim / MultichartAI, 2026-06-23
===============================================================================}

inputs:
    DMILen( 14 ),
    ADXThresh( 25 ),
    ATRMult( 3.0 ) ;

variables:
    _Crypto1MUSD( 0 ),
    ATRv( 0 ),
    diPlus( 0 ),
    diMinus( 0 ),
    adxv( 0 ),
    TrailLong( 0 ),
    TrailShort( 0 ) ;

{ ---- position size: ~$1M notional per trade, price-normalized ---- }
_Crypto1MUSD = Round( 1000000 / Close, 0 ) ;
if _Crypto1MUSD < 1 then
    _Crypto1MUSD = 1 ;

{ ---- indicators ---- }
ATRv    = AvgTrueRange( 14 ) ;
diPlus  = DMIPlus( DMILen ) ;
diMinus = DMIMinus( DMILen ) ;
adxv    = ADX( DMILen ) ;

{ ---- entries: ADX-gated DMI crossover, only when flat ---- }
if MarketPosition = 0 then
begin
    if adxv > ADXThresh and diPlus cross over diMinus then
        Buy ( "AdxLE" ) _Crypto1MUSD contract next bar at market ;
    if adxv > ADXThresh and diMinus cross over diPlus then
        SellShort ( "AdxSE" ) _Crypto1MUSD contract next bar at market ;
end ;

{ ---- ATR chandelier trailing stop: long ---- }
if MarketPosition = 1 then
begin
    if BarsSinceEntry = 0 then
        TrailLong = EntryPrice - ATRMult * ATRv
    else
        TrailLong = MaxList( TrailLong, Close - ATRMult * ATRv ) ;
    Sell ( "AdxLX" ) next bar at TrailLong stop ;
end ;

{ ---- ATR chandelier trailing stop: short ---- }
if MarketPosition = -1 then
begin
    if BarsSinceEntry = 0 then
        TrailShort = EntryPrice + ATRMult * ATRv
    else
        TrailShort = MinList( TrailShort, Close + ATRMult * ATRv ) ;
    BuyToCover ( "AdxSX" ) next bar at TrailShort stop ;
end ;
