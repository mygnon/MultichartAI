{===============================================================================
  SFJ_BBSqueeze_crypto
  -------------------------------------------------------------------------------
  PowerLanguage (EasyLanguage / MultiCharts64) trading strategy for crypto.
  Long + Short. Bollinger-Band SQUEEZE breakout (volatility-regime filter) + ATR trail.

  New concept vs the other Strategy-folder systems: a VOLATILITY-REGIME filter.
  Breakout STOP orders are placed ONLY while the Bollinger bands are COMPRESSED
  (band width below its own SqueezeLen-bar average = a "squeeze"/low-vol coil);
  the order then fills as price EXPANDS through the band. This targets the classic
  "volatility contraction -> expansion" move, rather than trading every breakout.
    - Direction: Buy at the upper band STOP / SellShort at the lower band STOP.
    - Exit: ATR chandelier trailing stop (the exit that worked across crypto trend
      strategies). Flat-only entry (no always-in flipping).
  Uses the built-in BollingerBand(Price, Length, NumDevs) (proven in _2021Basic_Osc).
  Sizing: _Crypto1MUSD = Round(1,000,000 / Close, 0) ~ $1M notional per trade.

  Inputs (all optimizable; names match the optimizer's ParamAxis exactly):
    BBLen      - Bollinger period
    BBmult     - Bollinger band width in std-devs
    SqueezeLen - lookback for the band-width average (squeeze reference)
    ATRMult    - trailing-stop distance in ATR(14) multiples

  Logic:
    UpBB=BollingerBand(C,BBLen,BBmult); DnBB=BollingerBand(C,BBLen,-BBmult)
    BWidth=UpBB-DnBB ;  squeeze = BWidth < Average(BWidth, SqueezeLen)
    while flat AND squeeze: Buy next bar at UpBB STOP ; SellShort next bar at DnBB STOP
    Exit  : ATR chandelier trail (ATRMult * ATR(14)) under/over running extreme

  Defaults: BBLen=20, BBmult=2.0, SqueezeLen=50, ATRMult=3.0
  Author: generated for Tim / MultichartAI, 2026-06-23
===============================================================================}

inputs:
    BBLen( 20 ),
    BBmult( 2.0 ),
    SqueezeLen( 50 ),
    ATRMult( 3.0 ) ;

variables:
    _Crypto1MUSD( 0 ),
    ATRv( 0 ),
    UpBB( 0 ),
    DnBB( 0 ),
    BWidth( 0 ),
    avgBW( 0 ),
    TrailLong( 0 ),
    TrailShort( 0 ) ;

{ ---- position size: ~$1M notional per trade, price-normalized ---- }
_Crypto1MUSD = Round( 1000000 / Close, 0 ) ;
if _Crypto1MUSD < 1 then
    _Crypto1MUSD = 1 ;

{ ---- Bollinger bands + squeeze (band width vs its own average) ---- }
ATRv   = AvgTrueRange( 14 ) ;
UpBB   = BollingerBand( Close, BBLen, BBmult ) ;
DnBB   = BollingerBand( Close, BBLen, -BBmult ) ;
BWidth = UpBB - DnBB ;
avgBW  = Average( BWidth, SqueezeLen ) ;

{ ---- entries: breakout STOP orders, only while squeezing + flat ---- }
if MarketPosition = 0 and BWidth < avgBW then
begin
    Buy ( "SqzLE" ) _Crypto1MUSD contract next bar at UpBB stop ;
    SellShort ( "SqzSE" ) _Crypto1MUSD contract next bar at DnBB stop ;
end ;

{ ---- ATR chandelier trailing stop: long ---- }
if MarketPosition = 1 then
begin
    if BarsSinceEntry = 0 then
        TrailLong = EntryPrice - ATRMult * ATRv
    else
        TrailLong = MaxList( TrailLong, Close - ATRMult * ATRv ) ;
    Sell ( "SqzLX" ) next bar at TrailLong stop ;
end ;

{ ---- ATR chandelier trailing stop: short ---- }
if MarketPosition = -1 then
begin
    if BarsSinceEntry = 0 then
        TrailShort = EntryPrice + ATRMult * ATRv
    else
        TrailShort = MinList( TrailShort, Close + ATRMult * ATRv ) ;
    BuyToCover ( "SqzSX" ) next bar at TrailShort stop ;
end ;
