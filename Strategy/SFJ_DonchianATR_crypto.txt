{===============================================================================
  SFJ_DonchianATR_crypto
  -------------------------------------------------------------------------------
  EasyLanguage (PowerLanguage / MultiCharts64) trading strategy for crypto.
  Long + Short. Donchian channel breakout entries + ATR trailing-stop exits.

  Rationale (from MultichartAI optimization research):
    - Crypto is strongly trend-driven; breakout/trend entries beat oscillators.
    - ATR trailing-stop exits HELP trend strategies (unlike reversal exits which
      churn and deepen drawdown). This strategy bakes the trailing exit in.
    - _Crypto1MUSD sizing => ~ $1,000,000 notional per trade (price-normalized,
      so NetProfit is comparable across BTC / ETH / BNB regardless of unit price).

  Inputs (all optimizable; match the optimizer's ParamAxis names exactly):
    Length      - Donchian lookback for the breakout channel (bars)
    ATRLength   - period of AvgTrueRange used for the trailing stop
    ATRMult     - trailing-stop distance in ATR multiples (wider = looser stop)

  Logic:
    Long  entry : Close > Highest(High, Length)[1]   -> Buy  next bar market
    Short entry : Close < Lowest (Low , Length)[1]   -> Sell next bar market
    Long  exit  : ATR chandelier trail under the highest close since entry
    Short exit  : ATR chandelier trail above the lowest  close since entry
  A new opposite breakout also flips the position (entries fire when not already
  in that direction), so the system is always-in once a trend starts.

  Defaults: Length=20, ATRLength=14, ATRMult=3.0
  Author: generated for Tim / MultichartAI, 2026-06-22
===============================================================================}

inputs:
    Length( 20 ),
    ATRLength( 14 ),
    ATRMult( 3.0 ) ;

variables:
    _Crypto1MUSD( 0 ),
    ATRVal( 0 ),
    HH( 0 ),
    LL( 0 ),
    TrailLong( 0 ),
    TrailShort( 0 ) ;

{ ---- position size: ~$1M notional per trade, price-normalized ---- }
_Crypto1MUSD = Round( 1000000 / Close, 0 ) ;
if _Crypto1MUSD < 1 then
    _Crypto1MUSD = 1 ;

{ ---- indicators ---- }
ATRVal = AvgTrueRange( ATRLength ) ;
HH     = Highest( High, Length )[1] ;   { prior-bar channel top    }
LL     = Lowest ( Low , Length )[1] ;   { prior-bar channel bottom }

{ ---- breakout entries (both directions) ---- }
if MarketPosition <> 1 and Close > HH then
    Buy ( "DonLE" ) _Crypto1MUSD contract next bar at market ;

if MarketPosition <> -1 and Close < LL then
    SellShort ( "DonSE" ) _Crypto1MUSD contract next bar at market ;

{ ---- ATR chandelier trailing stop: long ---- }
if MarketPosition = 1 then
begin
    if BarsSinceEntry = 0 then
        TrailLong = EntryPrice - ATRMult * ATRVal
    else
        TrailLong = MaxList( TrailLong, Close - ATRMult * ATRVal ) ;
    Sell ( "DonLX" ) next bar at TrailLong stop ;
end ;

{ ---- ATR chandelier trailing stop: short ---- }
if MarketPosition = -1 then
begin
    if BarsSinceEntry = 0 then
        TrailShort = EntryPrice + ATRMult * ATRVal
    else
        TrailShort = MinList( TrailShort, Close + ATRMult * ATRVal ) ;
    BuyToCover ( "DonSX" ) next bar at TrailShort stop ;
end ;
