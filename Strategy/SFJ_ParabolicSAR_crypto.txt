{===============================================================================
  SFJ_ParabolicSAR_crypto
  -------------------------------------------------------------------------------
  PowerLanguage (EasyLanguage / MultiCharts64) trading strategy for crypto.
  Long + Short. Wilder Parabolic SAR stop-and-reverse (self-contained reversal).

  Distinct mechanism from the other Strategy-folder systems (which all use an ATR
  chandelier trailing exit on top of a separate entry rule):
    - The Parabolic SAR IS both the trailing stop AND the entry: when price
      penetrates the SAR the position FLIPS (stop-and-reverse), so the system is
      always-in long or short. The SAR accelerates toward price (AF rises by
      AfStep each new extreme, capped at AfMax), tightening as a trend extends.
    - Only 2 parameters (AfStep, AfMax). Lower values = slower/looser SAR (fewer
      whipsaw flips, lets crypto trends run); higher = tighter/faster.
  Sizing: _Crypto1MUSD = Round(1,000,000 / Close, 0) ~ $1M notional per trade.

  Inputs (both optimizable; names match the optimizer's ParamAxis exactly):
    AfStep - acceleration-factor step (initial AF and per-new-extreme increment)
    AfMax  - acceleration-factor cap

  Logic (Welles Wilder PSAR):
    SAR_new = SAR + AF*(EP - SAR), clamped so it cannot penetrate the prior 2 bars'
    extreme; on penetration -> flip side, SAR=EP, EP=new extreme, AF=AfStep; else
    extend EP and raise AF by AfStep (cap AfMax). Flip -> Buy / SellShort at market.

  Defaults: AfStep=0.02, AfMax=0.20  (classic Wilder)
  Author: generated for Tim / MultichartAI, 2026-06-23
===============================================================================}

inputs:
    AfStep( 0.02 ),
    AfMax( 0.20 ) ;

variables:
    _Crypto1MUSD( 0 ),
    SAR( 0 ),
    EP( 0 ),
    AF( 0 ),
    pos( 0 ),       { internal trend state: 1 = long, -1 = short }
    newSAR( 0 ) ;

{ ---- position size: ~$1M notional per trade, price-normalized ---- }
_Crypto1MUSD = Round( 1000000 / Close, 0 ) ;
if _Crypto1MUSD < 1 then
    _Crypto1MUSD = 1 ;

{ ---- Parabolic SAR state machine ---- }
if CurrentBar = 1 then
begin
    pos = 1 ;
    SAR = Low ;
    EP  = High ;
    AF  = AfStep ;
end
else
begin
    newSAR = SAR + AF * ( EP - SAR ) ;

    if pos = 1 then
    begin
        { SAR may not exceed the prior two lows }
        if newSAR > Low[1] then
            newSAR = Low[1] ;
        if CurrentBar >= 3 and newSAR > Low[2] then
            newSAR = Low[2] ;

        if Low <= newSAR then
        begin               { penetrated -> flip to short }
            pos = -1 ;
            newSAR = EP ;
            EP = Low ;
            AF = AfStep ;
        end
        else
        begin
            if High > EP then
            begin
                EP = High ;
                AF = MinList( AF + AfStep, AfMax ) ;
            end ;
        end ;
    end
    else
    begin                   { pos = -1 (short) }
        { SAR may not fall below the prior two highs }
        if newSAR < High[1] then
            newSAR = High[1] ;
        if CurrentBar >= 3 and newSAR < High[2] then
            newSAR = High[2] ;

        if High >= newSAR then
        begin               { penetrated -> flip to long }
            pos = 1 ;
            newSAR = EP ;
            EP = High ;
            AF = AfStep ;
        end
        else
        begin
            if Low < EP then
            begin
                EP = Low ;
                AF = MinList( AF + AfStep, AfMax ) ;
            end ;
        end ;
    end ;

    SAR = newSAR ;
end ;

{ ---- orders: stop-and-reverse on PSAR side flip ---- }
if pos = 1 and MarketPosition <> 1 then
    Buy ( "SarLE" ) _Crypto1MUSD contract next bar at market ;
if pos = -1 and MarketPosition <> -1 then
    SellShort ( "SarSE" ) _Crypto1MUSD contract next bar at market ;
